class Solution{   
public:
    int median(vector<vector<int>> &arr, int r, int c){
        // code here        
        int mx=0,mn=INT_MAX;
        
        for(int i=0;i<r;i++){
            mn=min(mn,arr[i][0]);
            mx=max(mx,arr[i][c-1]);

        }
        
       
        
        int midPos = (r*c+1)/2;
        int num=0;
        
        while(mn<mx){
            
            int mid = (mx+mn)/2;
            num=0;
            
            for(int i=0;i<r;i++){
                num += upper_bound(arr[i].begin(),arr[i].end(),mid)-arr[i].begin() ;
            }
            
            if(num==midPos) return num;
            
            if(num<midPos) mn=mid+1;
            else mx=mid;
            
        }
        
        return mn;
    }
};
