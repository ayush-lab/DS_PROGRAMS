#include <bits/stdc++.h>
using namespace std;

class Solution{

  public:
  
  
	int minDifference(int arr[], int n)  { 
	    // Your code goes here
	    int range=0;
	    
	    for(int i=0;i<n;i++){
	        range+=arr[i];
	    }
	    
	    int half = range/2;
	    
	    vector<vector<bool>> dp(n+1,vector<bool>(half+1,false));
	
	    for(int i=0;i<=n;i++){
	        dp[i][0]=true;
	    }
	    for(int i=1;i<=half;i++){
	        dp[0][i]=false;
	    }
	    
	    for(int i=1;i<=n;i++){
	        for(int j=1;j<=half;j++){
	            if(arr[i-1] <=j){
	                dp[i][j] = dp[i-1][j-arr[i-1]] || dp[i-1][j];
	            }
	            else dp[i][j] = dp[i-1][j];
	        }
	    }
	    
	    int ans=0;
	    
	    for(int i=0;i<=half;i++){

	        if(dp[n][i]){
	            ans=i;
	        }
	    }
	    
	    range-=2*ans;
	    return range;
	} 
};
